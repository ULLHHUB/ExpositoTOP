117 public void calculateDistanceMatrix() {
118 for (int i = 0; i < this.nodes + 1; i++) {
119 for (int j = 0; j < this.nodes + 1; j++) {
120 if (i != j) {
121 double diffXs = this.x[i] - this.x[j];
122 double diffYs = this.y[i] - this.y[j];
123 this.distanceMatrix[i][j] = Math.sqrt(diffXs * diffXs + diffYs * diffYs);
124 this.distanceMatrix[j][i] = this.distanceMatrix[i][j];
