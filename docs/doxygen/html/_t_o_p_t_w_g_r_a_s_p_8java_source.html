49 Solution ← ∅;
50 Evaluate the incremental costs of the candidate elements;

52 Build the restricted candidate list (RCL);
53 Select an element s from the RCL at random;
54 Solution ← Solution ∪ {s};
55 Reevaluate the incremental costs;

242 // Inserción del cliente en la ruta return: cliente, ruta, predecesor, coste
243 this.solution.setPredecessor((int)candidateSelected[0], (int)candidateSelected[2]);

245 this.solution.setSuccessor((int)candidateSelected[2], (int)candidateSelected[0]);
246 this.solution.setPredecessor(this.solution.getSuccessor((int)candidateSelected[0]), (int)candidateSelected[0]);
247
248 // Actualización de las estructuras de datos y conteo a partir de la posición a insertar
249 double costInsertionPre = departureTimes.get((int)candidateSelected[1]).get((int)candidateSelected[2]);
250 ArrayList<Double> route = departureTimes.get((int)candidateSelected[1]);
251 int pre=(int)candidateSelected[2], suc=-1;
252 int depot = this.solution.getIndexRoute((int)candidateSelected[1]);
253 do {
254 suc = this.solution.getSuccessor(pre);
255 costInsertionPre += this.solution.getDistance(pre, suc);
256
257 if(costInsertionPre < this.solution.getProblem().getReadyTime(suc)) {
258 costInsertionPre = this.solution.getProblem().getReadyTime(suc);

264 pre = suc;
